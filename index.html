#include #include #include #include #include // Coloque aqui o nome e a senha do Wi-Fi const char* ssid = "R2D2"; const char* password = "RogueOne01*"; // Chave do ThingSpeak para enviar dados const char* writeAPIKey = "AAYOJ0HC2YRZZ9ZF"; Adafruit_BME280 bme; bool sensorEncontrado = false; void setup() { Serial.begin(115200); delay(1000); Serial.println("Iniciando sistema..."); // Tenta encontrar o sensor no endereço 0x76 if (bme.begin(0x76)) { sensorEncontrado = true; Serial.println("Sensor BME280 encontrado no endereco 0x76."); } // Se não encontrar, tenta no endereço 0x77 else if (bme.begin(0x77)) { sensorEncontrado = true; Serial.println("Sensor BME280 encontrado no endereco 0x77."); } else { Serial.println("ERRO: sensor BME280 nao encontrado."); Serial.println("Verifique os fios VIN, GND, SCL e SDA."); while (1); } WiFi.begin(ssid, password); Serial.println("Conectando ao Wi-Fi..."); while (WiFi.status() != WL_CONNECTED) { delay(1000); Serial.println("Tentando conectar..."); } Serial.println("Wi-Fi conectado!"); Serial.print("IP do ESP32: "); Serial.println(WiFi.localIP()); } void loop() { if (WiFi.status() == WL_CONNECTED && sensorEncontrado) { float temperatura = bme.readTemperature(); float umidade = bme.readHumidity(); float pressao = bme.readPressure() / 100.0F; String url = "http://api.thingspeak.com/update?api_key="; url += writeAPIKey; url += "&field1=" + String(temperatura, 1); url += "&field2=" + String(umidade, 1); url += "&field3=" + String(pressao, 1); HTTPClient http; http.begin(url); int httpCode = http.GET(); String resposta = http.getString(); Serial.println("------------------------------"); Serial.println("Enviando dados para ThingSpeak..."); Serial.print("HTTP Code: "); Serial.println(httpCode); Serial.print("Resposta ThingSpeak: "); Serial.println(resposta); Serial.print("Temperatura: "); Serial.print(temperatura, 1); Serial.println(" C"); Serial.print("Umidade: "); Serial.print(umidade, 1); Serial.println(" %"); Serial.print("Pressao: "); Serial.print(pressao, 1); Serial.println(" hPa"); http.end(); } else { Serial.println("Wi-Fi desconectado. Tentando reconectar..."); WiFi.begin(ssid, password); } delay(20000); }
